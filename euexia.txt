<script>

function showPage(name){

  document.querySelectorAll('.page').forEach(page=>{
    page.classList.remove('active');
  });

  document.getElementById('page-' + name).classList.add('active');

  window.scrollTo(0,0);

}

function filterSport(btn,sport){

  document.querySelectorAll('.filter-btn').forEach(button=>{
    button.classList.remove('active');
  });

  btn.classList.add('active');

  document.querySelectorAll('.feed-post').forEach(post=>{

    if(sport === 'all' || post.dataset.sport === sport){
      post.style.display='block';
    }else{
      post.style.display='none';
    }

  });

}

function handleRegister(){

  const name=document.getElementById('reg-name').value.trim();
  const email=document.getElementById('reg-email').value.trim();
  const pass=document.getElementById('reg-pass').value.trim();

  if(!name || !email || !pass){

    alert('Plotëso të gjitha fushat');
    return;

  }

  const users=JSON.parse(localStorage.getItem('users') || '[]');

  const exists=users.find(u=>u.email===email);

  if(exists){

    alert('Ky email ekziston');
    return;

  }

  const user={
    name,
    email,
    pass
  };

  users.push(user);

  localStorage.setItem('users',JSON.stringify(users));

  localStorage.setItem('loggedUser',JSON.stringify(user));

  document.getElementById('reg-success').classList.add('show');

  setTimeout(()=>{
    showPage('community');
  },1500);

}

function handleLogin(){

  const email=document.getElementById('login-email').value.trim();
  const pass=document.getElementById('login-pass').value.trim();

  const users=JSON.parse(localStorage.getItem('users') || '[]');

  const user=users.find(u=>u.email===email && u.pass===pass);

  if(user){

    localStorage.setItem('loggedUser',JSON.stringify(user));

    document.getElementById('login-success').classList.add('show');

    setTimeout(()=>{
      showPage('community');
    },1500);

  }else{

    alert('Email ose fjalëkalim gabim');

  }

}

function showPostForm(){

  const form=document.getElementById('post-form');

  if(form.style.display==='none'){
    form.style.display='block';
  }else{
    form.style.display='none';
  }

}

function addPost(){

  const title=document.getElementById('post-title').value.trim();
  const content=document.getElementById('post-content').value.trim();
  const sport=document.getElementById('post-sport').value;

  if(!title || !content){

    alert('Plotëso të gjitha fushat');
    return;

  }

  const loggedUser=JSON.parse(localStorage.getItem('loggedUser'));

  if(!loggedUser){

    alert('Duhet të kyçesh fillimisht');
    return;

  }

  const feed=document.getElementById('community-feed');

  const html=`

  <div class="feed-post" data-sport="${sport}">

    <div class="feed-post-inner">

      <div class="author">

        <div class="author-left">

          <div class="avatar">
            ${loggedUser.name.charAt(0)}
          </div>

          <div>
            <h4>${loggedUser.name}</h4>
            <p>${new Date().toLocaleDateString()}</p>
          </div>

        </div>

        <span class="post-sport">${sport}</span>

      </div>

      <div class="feed-post-title">
        ${title}
      </div>

      <p>${content}</p>

    </div>

    <div class="feed-post-footer">
      Postim nga përdoruesi
    </div>

  </div>

  `;

  feed.insertAdjacentHTML('afterbegin',html);

  document.getElementById('post-title').value='';
  document.getElementById('post-content').value='';

  document.getElementById('post-form').style.display='none';

}
